```py
class Solution:
    def add(self, a: int, b: int) -> int:
        x = 0xffffffff
        a, b = a & x, b & x
        while b:
            a, b = a^b & x, (a&b) << 1
        return a if a <= 0x7fffffff else ~(a^x)
```

```js
 var add = function(a, b) {
  let c = 0;

  while (b) {
      c = (a & b) << 1;
      a ^= b;
      b = c;
  }

  return a;
};
```
