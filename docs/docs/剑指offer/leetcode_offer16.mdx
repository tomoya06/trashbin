```py
class Solution:
    def myPow(self, x: float, n: int) -> float:
        self.pownRes = {
            0: 1,
            1: x,
        }
        def calcPow(m):
            if m in self.pownRes:
                return self.pownRes[m]
            res = calcPow(m//2) * calcPow(m//2) * calcPow(m%2)
            self.pownRes[m] = res
            return res
        
        calc = calcPow(abs(n))
        return calc if n >= 0 else 1/calc
```

```py
class Solution:
    def myPow(self, x: float, n: int) -> float:
        def calcPow(m):
            if m == 0:
                return 1
            if m == 1:
                return x
    
            half = calcPow(m//2)
            res = half * half
            if m%2 == 1:
                res *= x
            return res
        
        calc = calcPow(abs(n))
        return calc if n >= 0 else 1/calc
```

```py
class Solution:
    def myPow(self, x: float, n: int) -> float:
        nn = n if n >= 0 else -n
        res = 1
        bit = x
        while nn:
            if nn & 1:
                res *= bit
            bit *= bit
            nn //= 2
        return res if n >= 0 else 1.0/res
```
