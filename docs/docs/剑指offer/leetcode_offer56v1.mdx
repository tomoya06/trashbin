class Solution:
    def singleNumbers(self, nums: List[int]) -> List[int]:
        xy, m = 0, 1
        for num in nums:
            xy ^= num
        while m & xy == 0:
            m <<= 1
        x, y = 0, 0
        for num in nums:
            if num & m:
                x ^= num
            else:
                y ^= num
        
        return [x, y]
