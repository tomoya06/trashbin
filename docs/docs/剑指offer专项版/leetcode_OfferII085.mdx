class Solution {
    List<String> res;

    private void dfs(String ans, int left, int right) {
        if (left == 0 && right == 0) {
            res.add(ans);
            return;
        }

        if (left == right) {
            dfs(ans + '(', left-1, right);
        } else {
            if (left > 0) {
                dfs(ans + '(', left-1, right);
            }
            dfs(ans + ')', left, right-1);
        }
    }

    public List<String> generateParenthesis(int n) {
        res = new ArrayList<>();
        dfs("", n, n);
        return res;
    }
}
