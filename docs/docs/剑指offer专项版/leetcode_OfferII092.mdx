    const left1s = s.split('').map(() => 0);
    const nn = s.length;

    let cnt = 0;

        if (s[i] === '0') {
            cnt += 1;
        }
        right0s[i] = cnt;
    }
    cnt = 0;
    for (let i=0; i<nn; i+=1) {
        if (s[i] === '1') {
