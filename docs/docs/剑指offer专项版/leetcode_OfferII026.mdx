class Solution {
    public void reorderList(ListNode head) {
        if (head == null || head.next == null) {
            return;
        }

        ListNode p = head, q = head;
        while (q != null) {
            q = q.next;
            if (q != null) {
                q = q.next;
                p = p.next;
            }
        }

        p.next = null;

        while (p != null) {
            nxt = p.next;
            p.next = q;
            q = p;
            p = nxt;

            p = nxt;
        }
    }
}
