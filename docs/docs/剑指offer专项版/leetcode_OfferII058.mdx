class MyCalendar {

    private TreeMap<Integer, Integer> cal;

    public MyCalendar() {
        this.cal = new TreeMap<>();
    }
    
    public boolean book(int start, int end) {
        Integer prev = cal.floorKey(start);
        Integer next = cal.ceilingKey(start);

        if ((prev == null || cal.get(prev) <= start) && (next == null || next >= end)) {
            cal.put(start, end);
            return true;
        }
        return false;
    }
}
