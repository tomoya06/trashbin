```java
class Solution {
    public int numSubarrayProductLessThanK(int[] nums, int k) {
        int left = 0, right = 0;
        int res = 1;
        int ans = 0;
        while (right<nums.length) {
            res *= nums[right];
            while (res >= k && left <= right) {
                res /= nums[left];
                left += 1;
            }
            if (left <= right) {
                ans += right-left+1;
            }
            right += 1;
        }
        return ans;
    }
}
```
