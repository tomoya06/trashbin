class Solution {
    public int countSubstrings(String s) {
        int ans = 0, n = s.length(), ml, mr;
        for (int i=0; i<2*n-1; i+=1) {
            ml = i/2;
            mr = (i+1)/2;
            while (ml >= 0 && mr < n && s.charAt(ml) == s.charAt(mr)) {
                ml -= 1;
                mr += 1;
                ans += 1;
            }
        }

        return ans;
    }
}
