class Solution {
    public int searchInsert(int[] nums, int target) {
        int lo=0, hi=nums.length, mid;

        while (lo < hi) {
            mid = (lo+hi) / 2;
            if (nums[mid] >= target) {
                hi = mid;
            } else {
                lo = mid+1;
            }
        }

        return lo;
    }
}
