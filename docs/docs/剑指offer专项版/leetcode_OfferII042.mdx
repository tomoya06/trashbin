class RecentCounter {

    Deque<Integer> queue;

    public RecentCounter() {
        this.queue = new LinkedList<>();
    }
    
    public int ping(int t) {
        this.queue.add(t);

        while (this.queue.peek() < t-3000) {
            this.queue.poll();
        }

        return this.queue.size();
    }
}
